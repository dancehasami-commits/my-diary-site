const adminBtn = document.getElementById("adminToggleBtn"); // 状態読み込み let isAdmin = localStorage.getItem("isAdmin") === "true"; // 表示更新 function updateAdminBtn() { adminBtn.textContent = isAdmin ? "管理者モード: ON" : "管理者モード: OFF"; } updateAdminBtn(); // 切り替え adminBtn.addEventListener("click", () => { isAdmin = !isAdmin; localStorage.setItem("isAdmin", isAdmin); updateAdminBtn(); });
